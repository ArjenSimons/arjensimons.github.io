NeonFighter is a Bullet Hell game in which you progress through levels to fight enemies of increasing difficulty, ending with a boss fight. The focus while developing NeonFighter was to learn Unity, since it was the first game I made with the engine.

The following script lets a bullet explode once it reaches a certain position. When exploding it spawns an n amount of bullets in a circle around the exploding bullet.

using UnityEngine;

public class BulletExplode : MonoBehaviour {

    [SerializeField] private float explodingPoint;
    [SerializeField] private float bulletAmount;

    private float rotationChange;

    private void Start()
    {
        rotationChange = 360 / bulletAmount;
    }

    private void FixedUpdate()
    {
        if (transform.position.x < explodingPoint)
        {
            //Play audio effect
            AudioManager.instance.Play("BulletExplode");

            float rot = 0;

            //Spawn bullets
            for (int i = 0; i < bulletAmount; i++)
            {
                rot += rotationChange;
                Pooler.instance.Spawn(PoolerObject.BULLET, transform.position, Quaternion.Euler(0, 0, rot));
            }
            Destroy(gameObject);
        }
    }
}
